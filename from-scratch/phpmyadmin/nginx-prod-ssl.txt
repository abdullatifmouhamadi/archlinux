server {
	listen 80;
	server_name www.phpmyadmin.maoredev.fr phpmyadmin.maoredev.fr;
	return 301 https://$host$request_uri;
}


server {
    server_name www.phpmyadmin.maoredev.fr phpmyadmin.maoredev.fr;
    
    index index.php;
    
    access_log /var/log/nginx/phpmyadmin.access.log;
    error_log /var/log/nginx/phpmyadmin.error.log;

    root /usr/share/webapps/phpMyAdmin;
    location / {
        try_files $uri $uri/ =404;
    }

    error_page 404 /index.php;

    location ~ \.php$ {
        try_files $uri $document_root$fastcgi_script_name =404;

        fastcgi_split_path_info ^(.+\.php)(/.*)$;
        fastcgi_pass unix:/run/php-fpm/php-fpm.sock;
        fastcgi_index index.php;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        include fastcgi_params;

        fastcgi_param HTTP_PROXY "";
        fastcgi_param HTTPS on;
        fastcgi_request_buffering off;
   }
   
	# ACME challenge
	location ^~ /.well-known {
		allow all;
		alias /var/lib/letsencrypt/.well-known/;
		default_type "text/plain";
		try_files $uri =404;
	}

    # SSL
	listen 443 ssl;
	ssl_certificate           /etc/letsencrypt/live/phpmyadmin.maoredev.fr/fullchain.pem;
	ssl_certificate_key       /etc/letsencrypt/live/phpmyadmin.maoredev.fr/privkey.pem;
	ssl_trusted_certificate   /etc/letsencrypt/live/phpmyadmin.maoredev.fr/chain.pem;
}
